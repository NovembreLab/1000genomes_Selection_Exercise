### 1.3 Filtering Data and QC

#### Preparing CLST / IID Files 

```bash
# remove the first line and format file as IID | IID | CLST for CHB, YRI, CEU
sed '1d' integrated_call_samples_v3.20130502.ALL.panel | awk '($2 == "CHB" || $2 == "YRI" || $2 == "CEU") {print $1, $1, $2}' > chb_yri_ceu_clst.txt

# write a file of just IIDs 
awk '{print $1}' chb_yri_ceu_clst.txt > chb_yri_ceu_iids.txt

# write a clst file for just CHB and YRI 
awk '{($3 == "CHB" || $3 == "YRI") print $0}' chb_yri_ceu_clst.txt > chb_yri_clst.txt

# write a clst file for just CHB and CEU
awk '{if($3 == "CHB" || $3 == "CEU") print $0}' chb_yri_ceu_clst.txt > chb_ceu_clst.txt

# write a clst file for just YRI and CEU
awk  '($3 == "YRI" || $3 == "CEU") {print $0}' chb_yri_ceu_clst.txt > yri_ceu_clst.txt

# write a file of just IIDs from CHB 
awk '($3 == "CHB") {print $1}' chb_yri_ceu_clst.txt > chb_iids.txt

# check if # individuals are correct
wc -l chb_yri_ceu_iids.txt chb_yri_clst.txt chb_ceu_clst.txt yri_ceu_clst.txt chb_iids.txt
```

#### Preparing Genomic Data

```bash
# filter out indels (-V indels), multi-allelic sites (-m 2 -M 2), and keep individuals (-S) from YRI, CEU, CHB 
# this will be used for cross-population based selection analysis
bcftools view -V indels -m 2 -M 2 -S chb_yri_ceu_iids.txt -O z -q '.1 minor' 1000genomes_phase3_chr2_108013601_111013601.vcf.gz > chb_yri_ceu_chr2_108013601_111013601_filtered.vcf.gz

# keep YRI individuals from the filtered vcf
# this will be used for haplotype based selection analysis.
bcftools view -S chb_iids.txt -O z chb_yri_ceu_chr2_108013601_111013601_filtered.vcf.gz > chb_chr2_108013601_111013601_filtered.vcf.gz

# write bim file from vcf (-H supress header)
bcftools view chb_chr2_108013601_111013601_filtered.vcf.gz -H | awk '{print $1, $3, "0", $2, $4, $5}' > chr1_158000000_162000000_filtered.bim

# get genetic positions and interpolate genetic positions for variants not present in the genetic_map
plink --bim chr2_108013601_111013601_filtered.bim \
      --cm-map genetic_map_chr1_combined_b37.txt 1 \
      --make-just-bim \
      --out chr2_108013601_111013601_filtered_gm

# format as map file
awk '{print $1, $2, $3, $4}' chr2_108013601_111013601_filtered_gm.bim > chr2_108013601_111013601_filtered_gm.map 
```

## 2. Computing Selection Statistics 

### 2.1 Fst

```bash
# Fst between CHB and YRI
plink --vcf chb_yri_ceu_chr2_108013601_111013601_filtered.vcf.gz \
      --fst \
      --within chb_yri_clst.txt \
      --double-id \
      --id-delim . \
      --out yri_ceu_chr2_108013601_111013601_filtered \
      --set-missing-var-ids @:#
     
# Fst between CHB and CEU
plink --vcf chb_yri_ceu_chr2_108013601_111013601_filtered.vcf.gz \
      --fst \
      --within chb_ceu_clst.txt \
      --double-id \
      --id-delim . \
      --out chb_ceu_chr2_108013601_111013601_filtered \
      --set-missing-var-ids @:#
             
# Fst between YRI and CEU
plink --vcf chb_yri_ceu_chr2_108013601_111013601_filtered.vcf.gz \
      --fst \
      --within yri_ceu_clst.txt \
      --double-id \
      --id-delim . \
      --out ceu_chb_chr2_108013601_111013601_filtered \
      --set-missing-var-ids @:#
```
